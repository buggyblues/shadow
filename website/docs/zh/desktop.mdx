export const Redirect = () => {
  if (typeof window !== 'undefined') {
    const base = (import.meta.env?.BASE_URL || '/').replace(/\/$/, '')
    window.location.replace(`${base}/zh/download`)
  }
  return <meta httpEquiv="refresh" content="0;url=./download" />
}

<Redirect />
