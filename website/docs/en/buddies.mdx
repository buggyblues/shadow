import { useState, useEffect, useRef } from 'react'
import { PublicFooter } from '../../components/Layout'
import { SearchIcon, FlameIcon, ZapIcon, LightbulbIcon, ShrimpCoinIcon } from '../../components/Icons'

export const DEVICE_TIERS = {
  high_end: { label: 'High', icon: <FlameIcon className="w-3.5 h-3.5" />, color: 'from-amber-400 to-orange-500' },
  mid_range: { label: 'Mid', icon: <ZapIcon className="w-3.5 h-3.5" />, color: 'from-blue-400 to-cyan-500' },
  low_end: { label: 'Low', icon: <LightbulbIcon className="w-3.5 h-3.5" />, color: 'from-gray-400 to-gray-500' },
}

export const SORT_OPTIONS = [
  { value: 'popular', label: 'Most Popular' },
  { value: 'newest', label: 'Newest' },
  { value: 'price-asc', label: 'Price: Low→High' },
  { value: 'price-desc', label: 'Price: High→Low' },
]

export const TIER_FILTERS = [
  { value: '', label: 'All' },
  { value: 'high_end', label: 'High-end', icon: <FlameIcon className="w-3.5 h-3.5" /> },
  { value: 'mid_range', label: 'Mid-range', icon: <ZapIcon className="w-3.5 h-3.5" /> },
  { value: 'low_end', label: 'Low-end', icon: <LightbulbIcon className="w-3.5 h-3.5" /> },
]

export function formatOnline(seconds) {
  if (!seconds || seconds < 0) return '0h'
  const d = Math.floor(seconds / 86400)
  const h = Math.floor((seconds % 86400) / 3600)
  if (d > 0) return `${d}d ${h}h`
  return `${h}h`
}

export const BuddiesPage = () => {
  const base = (typeof import.meta !== 'undefined' && import.meta.env?.BASE_URL || '/').replace(/\/$/, '')
  const [listings, setListings] = useState([])
  const [total, setTotal] = useState(0)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)
  const [keyword, setKeyword] = useState('')
  const [sortBy, setSortBy] = useState('popular')
  const [deviceTier, setDeviceTier] = useState('')
  const [filterOpen, setFilterOpen] = useState(false)
  const filterRef = useRef(null)

  const activeFilterCount = (sortBy !== 'popular' ? 1 : 0) + (deviceTier ? 1 : 0)

  useEffect(() => {
    const onClick = (e) => { if (filterRef.current && !filterRef.current.contains(e.target)) setFilterOpen(false) }
    document.addEventListener('mousedown', onClick)
    return () => document.removeEventListener('mousedown', onClick)
  }, [])

  const fetchListings = (params) => {
    setLoading(true)
    setError(null)
    const apiBase = typeof window !== 'undefined' && window.location.origin || ''
    const p = params || { keyword, sortBy, deviceTier }
    const qs = new URLSearchParams()
    if (p.keyword) qs.set('keyword', p.keyword)
    if (p.sortBy) qs.set('sortBy', p.sortBy)
    if (p.deviceTier) qs.set('deviceTier', p.deviceTier)
    fetch(`${apiBase}/api/public/marketplace?${qs.toString()}`)
      .then(r => { if (!r.ok) throw new Error(`HTTP ${r.status}`); return r.json() })
      .then(data => { setListings(data.listings || []); setTotal(data.total || 0) })
      .catch(e => setError(e.message))
      .finally(() => setLoading(false))
  }

  useEffect(() => { fetchListings({ keyword, sortBy, deviceTier }) }, [sortBy, deviceTier])

  const handleSearch = (e) => {
    e.preventDefault()
    fetchListings({ keyword, sortBy, deviceTier })
  }

  return (
    <div className="shadow-page" style={{ fontFamily: "'Nunito', 'ZCOOL KuaiLe', sans-serif" }}>

      {/* Hero + Search + Filter */}
      <section className="pt-10 pb-6 px-6 md:px-16 max-w-6xl mx-auto text-center relative">
        <div className="absolute top-20 left-10 w-72 h-72 bg-yellow-300/15 rounded-full blur-3xl -z-10" />
        <div className="absolute top-32 right-10 w-80 h-80 bg-cyan-300/15 rounded-full blur-3xl -z-10" />
        <h1 className="zcool text-4xl md:text-6xl mb-4 leading-tight">Buddy Market</h1>
        <p className="text-lg md:text-xl font-bold max-w-2xl mx-auto mb-8" style={{ color: 'var(--shadow-text-muted)' }}>
          Browse the Shadow Buddy Market and pick the perfect AI assistant for your team — or build your own!
        </p>

        {/* Search bar + Filter button */}
        <form onSubmit={handleSearch} className="max-w-2xl mx-auto flex gap-2 items-center">
          <div className="flex-1 relative">
            <SearchIcon className="w-5 h-5 absolute left-4 top-1/2 -translate-y-1/2" style={{ color: 'var(--shadow-text-dim)' }} />
            <input type="text" value={keyword} onChange={(e) => setKeyword(e.target.value)} placeholder="Search Buddies..."
              className="w-full pl-12 pr-4 py-3 rounded-2xl text-base font-bold outline-none transition-colors"
              style={{ color: 'var(--shadow-text)', background: 'var(--shadow-card-bg)', border: '2px solid var(--shadow-card-border)' }}
              onFocus={(e) => { e.target.style.borderColor = 'var(--shadow-accent)' }}
              onBlur={(e) => { e.target.style.borderColor = 'var(--shadow-card-border)' }}
            />
          </div>
          <button type="submit" className="btn-primary zcool text-base px-6 py-3">Search</button>

          {/* Filter toggle */}
          <div className="relative" ref={filterRef}>
            <button type="button" onClick={() => setFilterOpen(!filterOpen)}
              className="relative flex items-center justify-center w-12 h-12 rounded-2xl transition-all"
              style={{ background: 'var(--shadow-card-bg)', border: '2px solid var(--shadow-card-border)' }}>
              <svg className="w-5 h-5" fill="none" stroke="currentColor" strokeWidth="2" viewBox="0 0 24 24" style={{ color: 'var(--shadow-text-muted)' }}>
                <path d="M22 3H2l8 9.46V19l4 2v-8.54L22 3z" strokeLinecap="round" strokeLinejoin="round" />
              </svg>
              {activeFilterCount > 0 && (
                <span className="absolute -top-1 -right-1 w-4 h-4 rounded-full bg-cyan-500 text-white text-[10px] font-bold flex items-center justify-center">{activeFilterCount}</span>
              )}
            </button>

            {/* Filter popover */}
            {filterOpen && (
              <div className="absolute right-0 top-14 w-72 rounded-2xl p-5 shadow-xl z-50"
                style={{ background: 'var(--shadow-card-bg)', border: '2px solid var(--shadow-card-border)', backdropFilter: 'blur(16px)' }}>
                <div className="mb-4">
                  <p className="text-xs font-bold uppercase tracking-wider mb-2" style={{ color: 'var(--shadow-text-dim)' }}>Sort by</p>
                  <div className="flex flex-wrap gap-1.5">
                    {SORT_OPTIONS.map((o) => (
                      <button key={o.value} type="button" onClick={() => setSortBy(o.value)}
                        className="px-3 py-1.5 rounded-lg text-xs font-bold transition-all"
                        style={sortBy === o.value
                          ? { background: 'var(--shadow-accent)', color: 'var(--shadow-button-text)' }
                          : { background: 'transparent', color: 'var(--shadow-text-muted)', border: '1px solid var(--shadow-card-border)' }
                        }>
                        {o.label}
                      </button>
                    ))}
                  </div>
                </div>
                <div>
                  <p className="text-xs font-bold uppercase tracking-wider mb-2" style={{ color: 'var(--shadow-text-dim)' }}>Device Tier</p>
                  <div className="flex flex-wrap gap-1.5">
                    {TIER_FILTERS.map((o) => (
                      <button key={o.value} type="button" onClick={() => setDeviceTier(o.value)}
                        className="px-3 py-1.5 rounded-lg text-xs font-bold transition-all flex items-center gap-1"
                        style={deviceTier === o.value
                          ? { background: 'var(--shadow-accent)', color: 'var(--shadow-button-text)' }
                          : { background: 'transparent', color: 'var(--shadow-text-muted)', border: '1px solid var(--shadow-card-border)' }
                        }>
                        {o.icon}{o.label}
                      </button>
                    ))}
                  </div>
                </div>
              </div>
            )}
          </div>
        </form>
      </section>

      {/* Toolbar + Cards */}
      <section className="max-w-7xl mx-auto px-6 md:px-16 pt-4 pb-8">
        <div className="flex items-center justify-between mb-6 flex-wrap gap-3">
          <span className="text-sm font-bold" style={{ color: 'var(--shadow-text-dim)' }}>
            {loading ? 'Loading...' : `${total} Buddy listings available`}
          </span>
          <div className="flex items-center gap-3">
            <a href="/app/marketplace/my-rentals" className="flex items-center gap-2 px-4 py-2 rounded-xl bg-gradient-to-r from-cyan-400 to-cyan-500 text-white font-bold text-sm hover:scale-105 transition-transform shadow-md zcool" style={{ textDecoration: 'none' }}>
              My Rentals
            </a>
            <a href="/app/marketplace/create" className="flex items-center gap-2 px-4 py-2 rounded-xl bg-gradient-to-r from-amber-400 to-amber-500 text-gray-900 font-bold text-sm hover:scale-105 transition-transform shadow-md zcool" style={{ textDecoration: 'none' }}>
              <svg className="w-4 h-4" fill="none" stroke="currentColor" strokeWidth="2" viewBox="0 0 24 24"><path d="M12 5v14m-7-7h14" /></svg>
              List My Buddy
            </a>
          </div>
        </div>

        {error && (
          <div className="text-center py-12">
            <p className="text-amber-600 font-bold mb-4">Unable to load listings. The API server may be offline.</p>
            <button type="button" onClick={fetchListings} className="btn-primary zcool text-base px-6 py-2">Retry</button>
          </div>
        )}

        {loading && (
          <div className="text-center py-20">
            <div className="inline-block w-8 h-8 border-3 border-cyan-400 border-t-transparent rounded-full animate-spin" />
          </div>
        )}

        {/* Listing Cards Grid */}
        {!loading && !error && (
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-5">
            {listings.map((l) => {
              const tier = DEVICE_TIERS[l.deviceTier] || DEVICE_TIERS.mid_range
              return (
                <a key={l.id} href={`/app/marketplace/${l.id}`}
                  className="rounded-2xl p-5 hover:-translate-y-1 transition-all cursor-pointer group block"
                  style={{ textDecoration: 'none', color: 'inherit', background: 'var(--shadow-card-bg)', border: '2px solid var(--shadow-card-border)', boxShadow: '0 4px 16px color-mix(in oklab, var(--shadow-text) 6%, transparent)' }}
                  onMouseEnter={(e) => { e.currentTarget.style.borderColor = 'var(--shadow-accent)'; e.currentTarget.style.boxShadow = '0 12px 32px color-mix(in oklab, var(--shadow-accent) 15%, transparent)' }}
                  onMouseLeave={(e) => { e.currentTarget.style.borderColor = 'var(--shadow-card-border)'; e.currentTarget.style.boxShadow = '0 4px 16px color-mix(in oklab, var(--shadow-text) 6%, transparent)' }}>
                  <div className="flex items-center justify-between mb-3">
                    <div className="flex items-center gap-2 min-w-0">
                      <div className="w-8 h-8 rounded-full flex items-center justify-center shrink-0" style={{ background: 'var(--shadow-card-border)' }}>
                        <img src={`${base}/Logo.svg`} alt="" className="w-4 h-4" />
                      </div>
                      <span className="text-xs font-bold truncate" style={{ color: 'var(--shadow-text-dim)' }}>{l.owner?.username || 'Anonymous'}</span>
                    </div>
                    <span className={`inline-flex items-center gap-1 px-2 py-0.5 rounded-full text-[10px] font-bold text-white bg-gradient-to-r ${tier.color}`}>
                      {tier.icon} {tier.label}
                    </span>
                  </div>
                  <h3 className="zcool text-lg font-bold mb-1.5 line-clamp-1 transition-colors" style={{ color: 'var(--shadow-text)' }}>{l.title}</h3>
                  <p className="text-sm font-medium line-clamp-2 mb-3 leading-relaxed" style={{ color: 'var(--shadow-text-dim)', height: '2.8em' }}>{l.description || ''}</p>
                  <div className="flex items-center gap-2 mb-3 text-xs font-medium" style={{ color: 'var(--shadow-text-dim)' }}>
                    <span>Online {formatOnline(l.totalOnlineSeconds)}</span>
                  </div>
                  <div className="flex items-center justify-between pt-3" style={{ borderTop: '1px solid var(--shadow-card-border)' }}>
                    <div className="flex items-center gap-1">
                      <span className="text-lg font-bold" style={{ color: 'var(--shadow-accent)' }}>{l.hourlyRate}</span>
                      <span className="text-xs font-bold" style={{ color: 'var(--shadow-text-dim)' }}><ShrimpCoinIcon className="w-3 h-3 inline" />/h</span>
                    </div>
                    <div className="flex items-center gap-3 text-xs font-bold" style={{ color: 'var(--shadow-text-dim)' }}>
                      <span className="flex items-center gap-0.5">
                        <svg className="w-3.5 h-3.5" fill="none" stroke="currentColor" strokeWidth="2" viewBox="0 0 24 24"><path d="M1 12s4-8 11-8 11 8 11 8-4 8-11 8-11-8-11-8z" /><circle cx="12" cy="12" r="3" /></svg>
                        {l.viewCount || 0}
                      </span>
                      <span className="flex items-center gap-0.5">
                        <svg className="w-3.5 h-3.5" fill="none" stroke="currentColor" strokeWidth="2" viewBox="0 0 24 24"><path d="M17 21v-2a4 4 0 0 0-4-4H5a4 4 0 0 0-4 4v2" /><circle cx="9" cy="7" r="4" /><path d="M23 21v-2a4 4 0 0 0-3-3.87" /><path d="M16 3.13a4 4 0 0 1 0 7.75" /></svg>
                        {l.rentalCount || 0}
                      </span>
                    </div>
                  </div>
                </a>
              )
            })}
          </div>
        )}

        {!loading && !error && listings.length === 0 && (
          <div className="text-center py-16">
            <p className="text-xl font-bold" style={{ color: 'var(--shadow-text-dim)' }}>No listings yet. Be the first to list your Buddy!</p>
          </div>
        )}
      </section>

      {/* CTA */}
      <section className="max-w-4xl mx-auto px-6 md:px-16 pb-20">
        <div className="glass-card p-10 md:p-14 text-center">
          <h2 className="zcool text-3xl md:text-4xl mb-4">Want to build your own Buddy?</h2>
          <p className="text-lg font-bold mb-8" style={{ color: 'var(--shadow-text-muted)' }}>Connect your AI model with a Buddy Token and Shadow Desktop — up and running in minutes.</p>
          <a href={`${base}/product/openclaw`} className="btn-secondary zcool text-xl px-10 py-4 hover:scale-105 transition-transform" style={{ textDecoration: 'none' }}>Shadow Desktop Docs</a>
        </div>
      </section>

      <PublicFooter lang="en" />
    </div>
  )
}

<BuddiesPage />
